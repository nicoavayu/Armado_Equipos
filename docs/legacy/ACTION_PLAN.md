```javascript
import { supabase } from '../lib/supabaseClient';

class SupabaseAPI {
  async fetchOne(table, filters) {
    const { data, error } = await supabase
      .from(table)
      .select('*')
      .match(filters)
      .single();
    
    if (error) throw error;
    return data;
  }

  async fetchMany(table, filters = {}, options = {}) {
    let query = supabase.from(table).select('*');
    
    Object.entries(filters).forEach(([key, value]) => {
      query = query.eq(key, value);
    });
    
    if (options.orderBy) {
      query = query.order(options.orderBy.column, { 
        ascending: options.orderBy.ascending ?? true 
      });
    }
    
    if (options.limit) {
      query = query.limit(options.limit);
    }
    
    const { data, error } = await query;
    if (error) throw error;
    return data || [];
  }

  async insert(table, data) {
    const { data: result, error } = await supabase
      .from(table)
      .insert(data)
      .select()
      .single();
    
    if (error) throw error;
    return result;
  }

  async update(table, filters, updates) {
    const { data, error } = await supabase
      .from(table)
      .update(updates)
      .match(filters)
      .select()
      .single();
    
    if (error) throw error;
    return data;
  }

  async delete(table, filters) {
    const { error } = await supabase
      .from(table)
      .delete()
      .match(filters);
    
    if (error) throw error;
  }
}

export const db = new SupabaseAPI();
```

```javascript
  console.error('[ERROR]', error);
```

```javascript
import { useCallback, useRef } from 'react';

export const useOptimizedCallback = (callback) => {
  const callbackRef = useRef(callback);
  
  // Actualizar ref en cada render
  callbackRef.current = callback;
  
  // Retornar callback estable
  return useCallback((...args) => {

  }, []);
};
```
